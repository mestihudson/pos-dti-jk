\providecommand{\resumo}[4]
{
  \small
  \textit{\textbf{#1.} #2}\\
  \textbf{\textit{#3}} #4\\
}

\providecommand{\narrativa}[3]
{
  \textbf{como um} \emph{#1}, \textbf{eu quero} \emph{#2}, \textbf{para} \emph{#3}
}

\providecommand{\codigo}[5]
{
  \begin{figure}[h]
    \centering
    \caption{#2}
    \lstinputlisting[#1]{#3}
    Fonte: {#5}
    \label{fig:#4}
  \end{figure}
}

\providecommand{\imagem}[5]
{
  \begin{figure}[h]
    \centering
    \caption{#2}
    \efbox{
      \makebox[\textwidth]{
        \includegraphics[#1]{#3}
      }
    }
    Fonte: #5
    \label{fig:#4}
  \end{figure}
}

\providecommand{\figref}[1]
{
  \textbf{Figura \ref{fig:#1}}
}
