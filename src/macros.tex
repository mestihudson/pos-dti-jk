\providecommand{\resumo}[4]{
  \small
  \textit{\textbf{#1.} #2}\\
  \textbf{\textit{#3}} #4\\
}

\providecommand{\narrativa}[3]{
  \textbf{como um} \emph{#1}, \textbf{eu quero} \emph{#2}, \textbf{para} \emph{#3}
}

\providecommand{\codigo}[5]{
  \begin{figure}[h]
    \centering
    \lstinputlisting[frame=none,#1]{#3}
    \caption{#2}
    \small{Fonte: {#5}}
    \label{fig:#4}
  \end{figure}
}

\providecommand{\imagem}[5]{
  \begin{figure}[h]
    \centering
    \efbox[rightline=false,topline=false,leftline=false,bottomline=false]{
      \makebox[\textwidth]{
        \includegraphics[#1]{#3}
      }
    }
    \caption{#2}
    \small{Fonte: #5}
    \label{fig:#4}
  \end{figure}
}

\providecommand{\figref}[1]{
  \textbf{Figura \ref{fig:#1}}
}

\providecommand{\citacaolonga}[2]{
  \begin{addmargin}[6cm]{0cm}
    \justify{
      \hspace*{.5cm} \small{#1}
    }
    #2
  \end{addmargin}
}
